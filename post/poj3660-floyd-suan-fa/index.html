POJ3660——Floyd算法

  1. 2021-03-18
  2. 265字
  3. 2 min read

Floyd算法的另一个用途，与传递闭包联系起来，当当这个点的入度和出度之和为n-1时，这个点的位置一定就确定了，因此只需要用Floyd算法计算出各个点之间的可达关系即可。

#include<iostream>
using namespace std;
const int INF = 0x3f3f3f3f;
int main()
{
	int n, m, field[105][105];
	cin >> n >> m;
	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= n; j++)
		{
			if (i == j)	field[i][j] = 0;
			else
			{
				field[i][j] = INF;
			}
		}
	}
	for (int i = 1; i <= m; i++)
	{
		int a, b;
		cin >> a >> b;
		field[a][b] = 1;
	}
	for (int i = 1; i <= n; i++)
	{
		for (int j = 1; j <= n; j++)
		{
			for (int k = 1; k <= n; k++)
			{
				if (field[j][k] > field[j][i] + field[i][k])
				{
					field[j][k] = field[j][i] + field[i][k];
				}
			}
		}
	}
    //在此之前是Floyd算法
	int num = 0;
	for (int i = 1; i <= n; i++)
	{
		int x=0, y=0;
		for (int j = 1; j <= n; j++)
		{//计算各点之间是否可达
			if (field[i][j] && field[i][j] != INF)	x++;
			if (field[j][i] && field[j][i] != INF)	y++;
		}
		if (x + y == n - 1)	num++;
	}
	cout << num;
}
